UTILISATEUR (
    id_utilisateur PK,
    email,
    mot_passe_hash,
    type_utilisateur,   // {ADMIN, ASSOCIATION}
    actif
)
ASSOCIATION (
    id_association PK,
    date_creation_association,
    nom_association,
    ufr,
    statut,             // {active, creation}
    email_contact,
    insta_contact,
    tel_contact,
    id_utilisateur FK UNIQUE → UTILISATEUR(id_utilisateur)
)
DIRIGEANT (
    id_dirigeant PK,
    prenom,
    nom,
    date_naissance,
    mail,
    tel,
    role,               // {Président, Trésorier, Secrétaire}
    date_debut_mandat,
    date_fin_mandat,
    id_association FK → ASSOCIATION(id_association)
)
TYPE_DOCUMENT (
    id_type_document PK,
    libelle,
    obligatoire,
    duree_validite_mois
)
DOCUMENT (
    id_document PK,
    nom_fichier,
    date_depot,
    date_expiration,
    statut,             // {Déposé, En validation, Validé, Refusé, Expiré}
    commentaire_refus,
    id_association FK → ASSOCIATION(id_association),
    id_type_document FK → TYPE_DOCUMENT(id_type_document)
)
NOTIFICATION (
    id_notification PK,
    date_envoi,
    sujet,
    message,
    type,               // {automatique, manuel}
    id_association FK → ASSOCIATION(id_association)
)
